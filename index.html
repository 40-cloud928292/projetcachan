Présentation du projet

Nous sommes un groupe de trois étudiants de l'IUT de Vélizy. Ce site regroupe l'avancement de notre projet d'automatisme et de programmation embarquée basé sur le cahier des charges technique du concours de Cachan.

L'objectif principal est de configurer les capteurs infrarouge de notre base Pololu 3Pi+ et de mettre au point des algorithmes de suivi de ligne pour franchir les différents circuits d'essais sans perte de trajectoire.

Parcours 3 - Filtrage et détection des tirets (Dashes)

Ce script assure le suivi de la ligne principale et comptabilise les lignes transversales. Un filtre numérique (anti-rebond par boucles) évite les doubles détections sur un même marquage. L'alignement repose sur un coefficient Kp de 0.06.

#include <Pololu3piPlus32U4.h>
using namespace Pololu3piPlus32U4;

LineSensors sensors;
Motors motors;

uint16_t s[5];
int baseSpeed = 110;
float Kp = 0.06;
int lastError = 0;

int noLineCount = 0;
const int NO_LINE_THRESHOLD = 25;
const int LINE_DETECT_MIN = 200;

int crossingCount = 0;
const int MAX_CROSSINGS = 4;
bool onDash = false;
bool waitingToStop = false;
int debounceCount = 0;
const int DEBOUNCE_LOOPS = 80;

#define DASH_SENSOR    s[4]
#define DASH_NEIGHBOR  s[3]
const int DASH_THRESHOLD = 500;

void setup()
{
  delay(1000);
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);

  int maxVal = 0;
  for (int i = 0; i < 5; i++)
    if (s[i] > maxVal) maxVal = s[i];

  if (maxVal < LINE_DETECT_MIN)
  {
    noLineCount++;
    if (noLineCount >= NO_LINE_THRESHOLD)
    {
      motors.setSpeeds(0, 0);
      while (1);
    }
  }
  else noLineCount = 0;

  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    bool dashAlone = (DASH_SENSOR > DASH_THRESHOLD) && (DASH_NEIGHBOR < DASH_THRESHOLD);

    if (dashAlone && !onDash)
    {
      onDash = true;
      crossingCount++;
      debounceCount = DEBOUNCE_LOOPS;

      if (crossingCount >= MAX_CROSSINGS)
        waitingToStop = true;
    }

    if (!dashAlone && onDash)
    {
      onDash = false;

      if (waitingToStop)
      {
        motors.setSpeeds(0, 0);
        while (1);
      }
    }
  }

  int error = (s[4] + s[3]) - (s[0] + s[1]);
  if (s[2] > (s[1] + s[3]) / 2)
    error = (s[3] - s[1]);
  if (abs(error - lastError) > 1500)
    error = lastError;

  lastError = error;

  int correction = error * Kp;
  int leftSpeed  = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;
  leftSpeed  = constrain(leftSpeed, -150, 200);
  rightSpeed = constrain(rightSpeed, -150, 200);
  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 13 - Demi-tour et retour

Le robot suit la ligne jusqu'à sa fin, puis effectue un demi-tour automatique (180°) pour revenir au point de départ. Utilise un algorithme de détection de perte de ligne et une fonction turn180() pour pivoter avec précision.

#include <Pololu3piPlus32U4.h>
using namespace Pololu3piPlus32U4;

LineSensors sensors;
Motors motors;

uint16_t s[5];

int baseSpeed = 110;
float Kp = 0.08;
float Kd = 0.04;
int lastError = 0;

int state = 0;  // 0 = aller, 1 = retour
int noLineCount = 0;
const int NO_LINE_THRESHOLD = 50;
const int LINE_DETECT_MIN = 200;

void turn180()
{
  motors.setSpeeds(0, 0);
  delay(200);
  motors.setSpeeds(100, -100);
  delay(1000);
  motors.setSpeeds(0, 0);
  delay(100);
  lastError = 0;
}

void setup()
{
  delay(1000);
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);

  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  // FIN ALLER → demi-tour
  if (state == 0)
  {
    if (maxVal < LINE_DETECT_MIN)
    {
      noLineCount++;
      if (noLineCount >= NO_LINE_THRESHOLD)
      {
        motors.setSpeeds(0, 0);
        delay(500);
        turn180();

        unsigned long startTime = millis();
        while (millis() - startTime < 800)
        {
          sensors.readCalibrated(s);
          motors.setSpeeds(110, 110);
        }

        noLineCount = 0;
        lastError = 0;
        state = 1;  // Passer au retour
        return;
      }
    }
    else
    {
      noLineCount = 0;
    }
  }

  // ===== suivi de ligne =====
  int error;

  if (s[2] > 500)
  {
    error = s[3] - s[1];
  }
  else
  {
    error = (s[4] + s[3]) - (s[0] + s[1]);
  }

  // filtrage des grosses erreurs
  if (abs(error - lastError) > 2000)
    error = lastError;

  int correction =
    (error * Kp) +
    ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  leftSpeed = constrain(leftSpeed, -150, 200);
  rightSpeed = constrain(rightSpeed, -150, 200);

  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 21 - Suivi de ligne simple

Programme de base pour le suivi de ligne utilisant un contrôleur PID (Proportionnel et Dérivé). Le robot utilise les 5 capteurs de réflectance pour ajuster sa trajectoire et rester centré sur la piste.

void loop()
{
  sensors.readCalibrated(s);

  // ========== SUIVI DE LIGNE SIMPLE ==========

  if (s[2] > 500)
  {

  else
  {

Parcours 22 - Gestion avancée des 3 vitesses (Rouge/Orange/Vert)

Parcours complexe avec gestion dynamique de 3 vitesses (SLOW=55, MEDIUM=90, FAST=130) selon 12 événements détectés. Machine d'états sophistiquée pour adapter la vitesse aux zones colorisées et aux intersections successives.

#include <Pololu3piPlus32U4.h>
using namespace Pololu3piPlus32U4;

LineSensors sensors;
Motors motors;

uint16_t s[5];

// Définition des 3 vitesses ajustées pour le parcours
int baseSpeed = 90;           // Vitesse initiale au départ (Orange = Moyenne)
const int SPEED_SLOW = 55;     // Rouge
const int SPEED_MEDIUM = 90;   // Orange
const int SPEED_FAST = 130;    // Vert

float Kp = 0.04;
float Kd = 0.08;
int lastError = 0;

// Détection de fin de ligne (Arrivée)
int noLineCount = 0;
const int NO_LINE_THRESHOLD = 25;
const int LINE_DETECT_MIN = 200;

// Gestion des repères et intersections
int crossingCount = 0;
bool onMarker = false;      
int debounceCount = 0;
const int DEBOUNCE_LOOPS = 50;

const int MARKER_THRESHOLD = 500;

void setup()
{
  delay(1000);
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);

  // 1. Sécurité et détection de la ligne d'arrivée
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  if (maxVal < LINE_DETECT_MIN)
  {
    noLineCount++;
    if (noLineCount >= NO_LINE_THRESHOLD)
    {
      motors.setSpeeds(0, 0);
      while (1);
    }
  }
  else
  {
    noLineCount = 0;
  }

  // 2. Machine d'états : Gestion des vitesses selon 12 événements
  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    bool detectLeft = (s[0] > MARKER_THRESHOLD);
    bool detectRight = (s[4] > MARKER_THRESHOLD);

    if ((detectLeft || detectRight) && !onMarker)
    {
      onMarker = true;
      crossingCount++;
      debounceCount = DEBOUNCE_LOOPS;

      // Application stricte des vitesses selon l'analyse du parcours
      switch (crossingCount)
      {
        case 1:  baseSpeed = SPEED_SLOW;   break;
        case 2:  baseSpeed = SPEED_FAST;   break;
        case 3:  baseSpeed = SPEED_SLOW;   break;
        case 4:  baseSpeed = SPEED_MEDIUM; break;
        case 5:  baseSpeed = SPEED_MEDIUM; break;
        case 6:  baseSpeed = SPEED_MEDIUM; break;
        case 7:  baseSpeed = SPEED_MEDIUM; break;
        case 8:  baseSpeed = SPEED_FAST;   break;
        case 9:  baseSpeed = SPEED_MEDIUM; break;
        case 10: baseSpeed = SPEED_MEDIUM; break;
        case 11: baseSpeed = SPEED_FAST;   break;
        case 12: baseSpeed = SPEED_SLOW;   break;
      }
    }

    if (!detectLeft && !detectRight && onMarker)
    {
      onMarker = false;
    }
  }

  // 3. Suivi de ligne PID avec priorité au capteur central
  int error = 0;
 
  if (s[2] > (s[1] + s[3]) / 2)
  {
    error = (s[3] - s[1]);
  }
  else
  {
    error = (s[4] + s[3]) - (s[0] + s[1]);
  }

  if (abs(error - lastError) > 1500)
  {
    error = lastError;
  }

  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  if (leftSpeed > 220)   leftSpeed = 220;
  if (leftSpeed < -150)  leftSpeed = -150;
  if (rightSpeed > 220)  rightSpeed = 220;
  if (rightSpeed < -150) rightSpeed = -150;

  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 23 - Détection de 11 tirets et virage 90° droite

Le robot compte les tirets détectés sur les capteurs latéraux. Au 11ème tiret détecté, il exécute automatiquement un virage de 90° vers la droite. Utilise des gains PID ajustés (Kp=0.1, Kd=0.05) pour un suivi précis en approche du virage.

#include <Pololu3piPlus32U4.h>
using namespace Pololu3piPlus32U4;

LineSensors sensors;
Motors motors;

uint16_t s[5];

int baseSpeed = 110;
float Kp = 0.1;
float Kd = 0.05;
int lastError = 0;

int markCount = 0;
bool onMark = false;
int debounceCount = 0;
const int DEBOUNCE_LOOPS = 120;
const int SIDE_THRESHOLD = 400;

bool turnDone = false;

void turn90Right()
{
  motors.setSpeeds(0, 0);
  delay(200);
  motors.setSpeeds(100, -100);
  delay(420);
  motors.setSpeeds(0, 0);
  delay(100);
  lastError = 0;
}

void setup()
{
  delay(1000);
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);

  // ========== COMPTER LES TIRETS ==========
  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    bool markDetected = (s[4] > SIDE_THRESHOLD) || (s[0] > SIDE_THRESHOLD) || (s[2] > 800);

    if (markDetected && !onMark)
    {
      onMark = true;
      markCount++;
      debounceCount = DEBOUNCE_LOOPS;

      // Au 11ème tiret → VIRAGE 90° DROITE
      if (markCount == 11 && !turnDone)
      {
        turn90Right();
        turnDone = true;
      }
    }

    if (!markDetected && onMark)
    {
      onMark = false;
    }
  }

  // ========== SUIVI DE LIGNE ==========
  int error = (s[3] + s[4]) - (s[0] + s[1]);

  if (abs(error - lastError) > 2000)
    error = lastError;

  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  leftSpeed = constrain(leftSpeed, -150, 200);
  rightSpeed = constrain(rightSpeed, -150, 200);

  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 9 - Repères latéraux et déviation à 90°

Le robot suit la ligne et détecte les marques sur les capteurs latéraux (s[0] ou s[4]). Au 7ème tiret latéral détecté, il exécute un virage de 90° vers la gauche (moteur gauche réduit). Utilise un anti-rebond pour éviter les doubles comptages.

{

    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

{
  sensors.readCalibrated(s);

  // --- Arrêt fin de piste ---
  int maxVal = 0;

        delay(100);

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  if (leftSpeed > 200) leftSpeed = 200;
  if (leftSpeed < -150) leftSpeed = -150;
  if (rightSpeed > 200) rightSpeed = 200;
  if (rightSpeed < -150) rightSpeed = -150;

  motors.setSpeeds(leftSpeed, rightSpeed);
}
