DÉVELOPPEMENT ROBOTIQUE

Structure, calibration de capteurs et tests en boucle fermée sur notre robot suiveur de ligne Pololu 3Pi+.

Présentation du projet

Nous sommes un groupe de trois étudiants de l'IUT de Vélizy. Ce site regroupe l'avancement de notre projet d'automatisme et de programmation embarquée basé sur le cahier des charges technique du concours de Cachan.

L'objectif principal est de configurer les capteurs infrarouge de notre base Pololu 3Pi+ et de mettre au point des algorithmes de suivi de ligne pour franchir les différents circuits d'essais sans perte de trajectoire.

Parcours 3 - Filtrage et détection des tirets (Dashes)

Ce script assure le suivi de la ligne principale et comptabilise les lignes transversales. Un filtre numérique (anti-rebond par boucles) évite les doubles détections sur un même marquage. L'alignement repose sur un coefficient Kp de 0.06.

#include <Pololu3piPlus32U4.h>   // Inclut la bibliothèque matérielle du robot
using namespace Pololu3piPlus32U4; // Utilise l'espace de noms pour simplifier le code

LineSensors sensors;  // Capteurs de ligne infrarouge
Motors motors;        // Moteurs du robot

uint16_t s[5];            // Tableau des 5 valeurs des capteurs
int baseSpeed = 110;      // Vitesse de base des moteurs
float Kp = 0.06;          // Gain proportionnel du correcteur
int lastError = 0;        // Dernière erreur mesurée (pour filtrage)

int noLineCount = 0;                // Compteur de cycles sans ligne détectée
const int NO_LINE_THRESHOLD = 25;   // Seuil avant arrêt (perte de ligne)
const int LINE_DETECT_MIN = 200;    // Valeur minimale pour considérer une ligne présente

int crossingCount = 0;           // Nombre de tirets détectés
const int MAX_CROSSINGS = 4;     // Nombre de tirets avant arrêt
bool onDash = false;             // Indique si on est actuellement sur un tiret
bool waitingToStop = false;      // Attend l'arrêt après le dernier tiret
int debounceCount = 0;           // Compteur d'anti-rebond
const int DEBOUNCE_LOOPS = 80;   // Nombre de cycles d'anti-rebond

#define DASH_SENSOR    s[4]      // Capteur utilisé pour détecter le tiret
#define DASH_NEIGHBOR  s[3]      // Capteur voisin pour confirmer le tiret
const int DASH_THRESHOLD = 500;  // Seuil de détection d'un tiret

void setup()
{
  delay(1000);  // Pause d'une seconde avant initialisation
  // Calibration automatique des capteurs en tournant sur place
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60); // Alternance de direction
    sensors.calibrate();  // Lecture pour calibration
  }
  motors.setSpeeds(0, 0); // Arrêt des moteurs après calibration
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture des valeurs calibrées des capteurs

  // Recherche de la valeur maximale parmi les capteurs
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
    if (s[i] > maxVal) maxVal = s[i];

  // Vérification de la perte de ligne
  if (maxVal < LINE_DETECT_MIN)
  {
    noLineCount++;                           // Incrémente le compteur de perte
    if (noLineCount >= NO_LINE_THRESHOLD)    // Si trop de cycles sans ligne
    {
      motors.setSpeeds(0, 0);  // Arrêt des moteurs
      while (1);               // Boucle infinie (arrêt définitif)
    }
  }
  else noLineCount = 0;  // Réinitialise le compteur si la ligne est retrouvée

  // Gestion de l'anti-rebond pour la détection des tirets
  if (debounceCount > 0)
  {
    debounceCount--;  // Décrémente le compteur d'anti-rebond
  }
  else
  {
    // Détection d'un tiret isolé sur le capteur droit
    bool dashAlone = (DASH_SENSOR > DASH_THRESHOLD) && (DASH_NEIGHBOR < DASH_THRESHOLD);

    if (dashAlone && !onDash)       // Si nouveau tiret détecté
    {
      onDash = true;                // On est sur un tiret
      crossingCount++;              // Incrémente le compteur de tirets
      debounceCount = DEBOUNCE_LOOPS; // Lance l'anti-rebond

      if (crossingCount >= MAX_CROSSINGS)  // Si on a atteint le nombre voulu
        waitingToStop = true;              // Prépare l'arrêt
    }

    if (!dashAlone && onDash)  // Si on vient de quitter le tiret
    {
      onDash = false;  // On n'est plus sur un tiret

      if (waitingToStop)  // Si on devait s'arrêter après ce tiret
      {
        motors.setSpeeds(0, 0);  // Arrêt des moteurs
        while (1);               // Boucle infinie
      }
    }
  }

  // Calcul de l'erreur de position par rapport à la ligne
  int error = (s[4] + s[3]) - (s[0] + s[1]);  // Différence capteurs droits - gauches
  if (s[2] > (s[1] + s[3]) / 2)               // Si le capteur central voit bien la ligne
    error = (s[3] - s[1]);                    // Erreur plus fine avec les capteurs adjacents
  if (abs(error - lastError) > 1500)          // Filtrage des variations brutales
    error = lastError;                        // Ignore la variation anormale

  lastError = error;  // Mémorise l'erreur pour le prochain cycle

  // Correction proportionnelle
  int correction = error * Kp;
  int leftSpeed  = baseSpeed + correction;   // Moteur gauche
  int rightSpeed = baseSpeed - correction;   // Moteur droit
  leftSpeed  = constrain(leftSpeed, -150, 200);  // Limite les vitesses
  rightSpeed = constrain(rightSpeed, -150, 200);
  motors.setSpeeds(leftSpeed, rightSpeed);   // Applique les vitesses
}

Parcours 13 - Demi-tour et retour

Le robot suit la ligne jusqu'à sa fin, puis effectue un demi-tour automatique (180°) pour revenir au point de départ. Utilise un algorithme de détection de perte de ligne et une fonction turn180() pour pivoter avec précision.

#include <Pololu3piPlus32U4.h>   // Bibliothèque du robot
using namespace Pololu3piPlus32U4; // Espace de noms

LineSensors sensors;  // Capteurs de ligne
Motors motors;        // Moteurs

uint16_t s[5];            // Valeurs des capteurs

int baseSpeed = 110;      // Vitesse de base
float Kp = 0.08;          // Gain proportionnel
float Kd = 0.04;          // Gain dérivé
int lastError = 0;        // Dernière erreur

int state = 0;                // 0 = aller, 1 = retour
int noLineCount = 0;          // Compteur absence de ligne
const int NO_LINE_THRESHOLD = 50;  // Seuil de perte de ligne
const int LINE_DETECT_MIN = 200;   // Seuil de détection

void turn180()
{
  motors.setSpeeds(0, 0);  // Arrêt avant la rotation
  delay(200);              // Pause de stabilisation
  motors.setSpeeds(100, -100);  // Rotation sur place
  delay(1000);                  // Durée du demi-tour (ajuster selon le robot)
  motors.setSpeeds(0, 0);  // Arrêt
  delay(100);              // Pause
  lastError = 0;           // Réinitialise l'erreur
}

void setup()
{
  delay(1000);  // Pause initiale
  // Calibration par rotation
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture capteurs

  // Détection du maximum
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  // FIN ALLER → demi-tour
  if (state == 0)  // Si on est en phase aller
  {
    if (maxVal < LINE_DETECT_MIN)  // Si la ligne est perdue
    {
      noLineCount++;
      if (noLineCount >= NO_LINE_THRESHOLD)  // Confirmation de la perte
      {
        motors.setSpeeds(0, 0);  // Arrêt
        delay(500);
        turn180();               // Demi-tour

        // Avance un peu pour retrouver la ligne
        unsigned long startTime = millis();
        while (millis() - startTime < 800)
        {
          sensors.readCalibrated(s);
          motors.setSpeeds(110, 110);  // Avance tout droit
        }

        noLineCount = 0;   // Réinitialise
        lastError = 0;
        state = 1;         // Passe en mode retour
        return;
      }
    }
    else
    {
      noLineCount = 0;  // Ligne retrouvée
    }
  }

  // ===== suivi de ligne =====
  int error;

  if (s[2] > 500)                    // Si le capteur central est bien sur la ligne
  {
    error = s[3] - s[1];             // Erreur basée sur les capteurs adjacents
  }
  else
  {
    error = (s[4] + s[3]) - (s[0] + s[1]); // Erreur basée sur les capteurs extérieurs
  }

  // Filtrage des variations brutales
  if (abs(error - lastError) > 2000)
    error = lastError;

  // Correction PID
  int correction =
    (error * Kp) +
    ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  leftSpeed = constrain(leftSpeed, -150, 200);
  rightSpeed = constrain(rightSpeed, -150, 200);

  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 21 - Suivi de ligne simple

Programme de base pour le suivi de ligne utilisant un contrôleur PID (Proportionnel et Dérivé). Le robot utilise les 5 capteurs de réflectance pour ajuster sa trajectoire et rester centré sur la piste.

#include <Pololu3piPlus32U4.h>   // Bibliothèque du robot
using namespace Pololu3piPlus32U4; // Espace de noms

LineSensors sensors;  // Capteurs de ligne
Motors motors;        // Moteurs

float Kd = 0.04;      // Gain dérivé

{

  // Calibration par rotation

  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture capteurs

  // ========== SUIVI DE LIGNE SIMPLE ==========
  int error;

  if (s[2] > 500)  // Si le capteur central est bien sur la ligne
  {
    // Ligne bien centrée : erreur basée sur les capteurs adjacents
    error = (s[3] - s[1]);
  }
  else
  {
    // Ligne sur les côtés : erreur basée sur les capteurs extérieurs

}

Parcours 22 - Gestion avancée des 3 vitesses (Rouge/Orange/Vert)

Parcours complexe avec gestion dynamique de 3 vitesses selon 12 événements détectés. Machine d'états sophistiquée pour adapter la vitesse aux zones et intersections.

#include <Pololu3piPlus32U4.h>   // Bibliothèque du robot
using namespace Pololu3piPlus32U4; // Espace de noms

LineSensors sensors;  // Capteurs de ligne
Motors motors;        // Moteurs

uint16_t s[5];            // Valeurs des capteurs

int baseSpeed = 90;       // Vitesse initiale
const int SPEED_SLOW = 55;    // Vitesse lente (rouge)
const int SPEED_MEDIUM = 90;  // Vitesse moyenne (orange)
const int SPEED_FAST = 130;   // Vitesse rapide (vert)

float Kp = 0.04;          // Gain proportionnel
float Kd = 0.08;          // Gain dérivé
int lastError = 0;        // Dernière erreur

int noLineCount = 0;                // Compteur perte de ligne
const int NO_LINE_THRESHOLD = 25;   // Seuil de perte
const int LINE_DETECT_MIN = 200;    // Seuil de détection ligne

int crossingCount = 0;           // Nombre d'événements détectés
bool onMarker = false;           // Indique si on est sur un marqueur
int debounceCount = 0;           // Anti-rebond
const int DEBOUNCE_LOOPS = 50;   // Cycles d'anti-rebond

const int MARKER_THRESHOLD = 500; // Seuil de détection d'un marqueur latéral

void setup()
{
  delay(1000);  // Pause initiale
  // Calibration
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture capteurs

  // Vérification présence ligne
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  if (maxVal < LINE_DETECT_MIN)  // Ligne absente
  {
    noLineCount++;
    if (noLineCount >= NO_LINE_THRESHOLD)  // Perte confirmée
    {
      motors.setSpeeds(0, 0);  // Arrêt
      while (1);               // Boucle infinie
    }
  }
  else
  {
    noLineCount = 0;  // Ligne présente
  }

  // Gestion des marqueurs avec anti-rebond
  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    bool detectLeft = (s[0] > MARKER_THRESHOLD);   // Marqueur à gauche
    bool detectRight = (s[4] > MARKER_THRESHOLD);  // Marqueur à droite

    if ((detectLeft || detectRight) && !onMarker)  // Nouveau marqueur détecté
    {
      onMarker = true;                 // On est sur un marqueur
      crossingCount++;                 // Incrémente le compteur
      debounceCount = DEBOUNCE_LOOPS;  // Anti-rebond

      // Changement de vitesse selon le numéro de l'événement
      switch (crossingCount)
      {
        case 1:  baseSpeed = SPEED_SLOW;   break;  // Événement 1 → lent
        case 2:  baseSpeed = SPEED_FAST;   break;  // Événement 2 → rapide
        case 3:  baseSpeed = SPEED_SLOW;   break;  // etc.
        case 4:  baseSpeed = SPEED_MEDIUM; break;
        case 5:  baseSpeed = SPEED_MEDIUM; break;
        case 6:  baseSpeed = SPEED_MEDIUM; break;
        case 7:  baseSpeed = SPEED_MEDIUM; break;
        case 8:  baseSpeed = SPEED_FAST;   break;
        case 9:  baseSpeed = SPEED_MEDIUM; break;
        case 10: baseSpeed = SPEED_MEDIUM; break;
        case 11: baseSpeed = SPEED_FAST;   break;
        case 12: baseSpeed = SPEED_SLOW;   break;
      }
    }

    if (!detectLeft && !detectRight && onMarker)  // Marqueur quitté
    {
      onMarker = false;
    }
  }

  // Calcul de l'erreur de suivi de ligne
  int error = 0;
 
  if (s[2] > (s[1] + s[3]) / 2)  // Ligne centrée
  {
    error = (s[3] - s[1]);       // Erreur fine
  }
  else
  {
    error = (s[4] + s[3]) - (s[0] + s[1]); // Erreur large
  }

  // Filtrage des variations brutales
  if (abs(error - lastError) > 1500)
  {
    error = lastError;
  }

  // Correction PID
  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  // Limitation des vitesses avec des bornes adaptées
  if (leftSpeed > 220)   leftSpeed = 220;
  if (leftSpeed < -150)  leftSpeed = -150;
  if (rightSpeed > 220)  rightSpeed = 220;
  if (rightSpeed < -150) rightSpeed = -150;

  motors.setSpeeds(leftSpeed, rightSpeed);  // Applique les vitesses
}

Parcours 23 - Détection de 11 tirets et virage 90° droite

Le robot compte les tirets détectés et exécute automatiquement un virage de 90° vers la droite au 11ème tiret.

#include <Pololu3piPlus32U4.h>   // Bibliothèque du robot
using namespace Pololu3piPlus32U4; // Espace de noms

LineSensors sensors;  // Capteurs de ligne
Motors motors;        // Moteurs

uint16_t s[5];            // Valeurs des capteurs

int baseSpeed = 110;      // Vitesse de base
float Kp = 0.1;           // Gain proportionnel
float Kd = 0.05;          // Gain dérivé
int lastError = 0;        // Dernière erreur

int markCount = 0;             // Compteur de marques
bool onMark = false;           // Indique si on est sur une marque
int debounceCount = 0;         // Anti-rebond
const int DEBOUNCE_LOOPS = 120; // Cycles d'anti-rebond
const int SIDE_THRESHOLD = 400; // Seuil de détection d'un tiret latéral

bool turnDone = false;  // Indique si le virage a déjà été effectué

void turn90Right()
{
  motors.setSpeeds(0, 0);     // Arrêt
  delay(200);                 // Pause
  motors.setSpeeds(100, -100); // Rotation à droite
  delay(420);                 // Durée pour ~90°
  motors.setSpeeds(0, 0);     // Arrêt
  delay(100);                 // Pause
  lastError = 0;              // Réinitialise l'erreur
}

void setup()
{
  delay(1000);  // Pause initiale
  // Calibration
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture capteurs

  // ========== COMPTER LES TIRETS ==========
  if (debounceCount > 0)
  {
    debounceCount--;  // Décrémente l'anti-rebond
  }
  else
  {
    // Détection d'un tiret (capteur droit, gauche ou central très fort)
    bool markDetected = (s[4] > SIDE_THRESHOLD) || (s[0] > SIDE_THRESHOLD) || (s[2] > 800);

    if (markDetected && !onMark)  // Nouveau tiret
    {
      onMark = true;                // On est sur un tiret
      markCount++;                  // Incrémente le compteur
      debounceCount = DEBOUNCE_LOOPS; // Anti-rebond

      if (markCount == 11 && !turnDone)  // Au 11ème tiret
      {
        turn90Right();    // Virage 90° droite
        turnDone = true;  // Virage effectué
      }
    }

    if (!markDetected && onMark)  // Tiret quitté
    {
      onMark = false;
    }
  }

  // ========== SUIVI DE LIGNE ==========
  int error = (s[3] + s[4]) - (s[0] + s[1]); // Erreur basée sur les capteurs extérieurs

  // Filtrage des variations brutales
  if (abs(error - lastError) > 2000)
    error = lastError;

  // Correction PID
  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  // Limitation des vitesses
  leftSpeed = constrain(leftSpeed, -150, 200);
  rightSpeed = constrain(rightSpeed, -150, 200);

  motors.setSpeeds(leftSpeed, rightSpeed);  // Applique
}

Parcours 5 - Demi-tour au 2ème tiret droit

  sensors.readCalibrated(s);  // Lecture capteurs

  // Vérification ligne
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  if (maxVal < LINE_DETECT_MIN)
  {
    noLineCount++;
    if (noLineCount >= NO_LINE_THRESHOLD)
    {
      motors.setSpeeds(0, 0);
      while (1);
    }
  }
  else
  {
    noLineCount = 0;
  }

  // Détection des tirets avec anti-rebond
  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    // Tiret détecté uniquement sur le capteur droit (s[4]) et pas sur le voisin (s[3])
    bool dashAlone = (s[4] > DASH_THRESHOLD) && (s[3] < DASH_THRESHOLD);

    if (dashAlone && !onDash)  // Nouveau tiret
    {
      onDash = true;
      crossingCount++;
      debounceCount = DEBOUNCE_LOOPS;

      if (crossingCount == 2 && !turnaroundDone)  // Au 2ème tiret
      {
        turnaroundDone = true;
        motors.setSpeeds(0, 0);   // Arrêt
        delay(200);
        motors.setSpeeds(50, -50); // Demi-tour
        delay(750);
        motors.setSpeeds(0, 0);   // Arrêt
        delay(100);
        lastError = 0;            // Réinitialise l'erreur
      }
    }

    if (!dashAlone && onDash)  // Fin du tiret
    {
      onDash = false;
    }
  }

  // Suivi de ligne
  int error = (s[4] + s[3]) - (s[0] + s[1]);
  if (s[2] > (s[1] + s[3]) / 2)
  {
    error = (s[3] - s[1]);
  }
  if (abs(error - lastError) > 1500)
  {
    error = lastError;
  }

  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  if (leftSpeed > 200) leftSpeed = 200;
  if (leftSpeed < -150) leftSpeed = -150;
  if (rightSpeed > 200) rightSpeed = 200;
  if (rightSpeed < -150) rightSpeed = -150;

  motors.setSpeeds(leftSpeed, rightSpeed);
}

Parcours 9 - Repères latéraux et déviation à 90°

Le robot détecte les marques sur les capteurs latéraux. Au 7ème tiret latéral détecté, il exécute un virage de 90° vers la gauche.

#include <Pololu3piPlus32U4.h>   // Bibliothèque du robot
using namespace Pololu3piPlus32U4; // Espace de noms

LineSensors sensors;  // Capteurs de ligne
Motors motors;        // Moteurs

uint16_t s[5];            // Valeurs des capteurs
int baseSpeed = 110;      // Vitesse de base
float Kp = 0.04;          // Gain proportionnel
float Kd = 0.08;          // Gain dérivé
int lastError = 0;        // Dernière erreur

int noLineCount = 0;                // Compteur perte de ligne
const int NO_LINE_THRESHOLD = 25;   // Seuil perte
const int LINE_DETECT_MIN = 200;    // Seuil détection

int sideCount = 0;               // Compteur de marques latérales
const int TURN_AT = 7;           // Numéro de la marque pour tourner
bool onMark = false;             // Sur une marque
int debounceCount = 0;           // Anti-rebond
const int DEBOUNCE_LOOPS = 80;   // Cycles d'anti-rebond
bool turnDone = false;           // Virage déjà effectué

const int SIDE_THRESHOLD = 500;  // Seuil marque latérale

void setup()
{
  delay(1000);  // Pause
  // Calibration
  for (int i = 0; i < 80; i++)
  {
    motors.setSpeeds((i < 40) ? 60 : -60, (i < 40) ? -60 : 60);
    sensors.calibrate();
  }
  motors.setSpeeds(0, 0);
}

void loop()
{
  sensors.readCalibrated(s);  // Lecture capteurs

  // Vérification ligne
  int maxVal = 0;
  for (int i = 0; i < 5; i++)
  {
    if (s[i] > maxVal) maxVal = s[i];
  }

  if (maxVal < LINE_DETECT_MIN)
  {
    noLineCount++;
    if (noLineCount >= NO_LINE_THRESHOLD)
    {
      motors.setSpeeds(0, 0);
      while (1);
    }
  }
  else
  {
    noLineCount = 0;
  }

  // Détection des marques latérales avec anti-rebond
  if (debounceCount > 0)
  {
    debounceCount--;
  }
  else
  {
    // Marque détectée à gauche (s[0]) ou à droite (s[4])
    bool markDetected = (s[4] > SIDE_THRESHOLD) || (s[0] > SIDE_THRESHOLD);

    if (markDetected && !onMark)  // Nouvelle marque
    {
      onMark = true;
      sideCount++;
      debounceCount = DEBOUNCE_LOOPS;

      if (sideCount == TURN_AT && !turnDone)  // 7ème marque → virage
      {
        turnDone = true;
        motors.setSpeeds(0, 0);   // Arrêt
        delay(200);

        motors.setSpeeds(50, -50); // Virage à gauche (inverser les signes pour droite)
        delay(375);                // Durée pour ~90°
        motors.setSpeeds(0, 0);   // Arrêt
        delay(100);

        lastError = 0;            // Réinitialise l'erreur
      }
    }

    if (!markDetected && onMark)  // Marque quittée
    {
      onMark = false;
    }
  }

  // Suivi de ligne
  int error = (s[4] + s[3]) - (s[0] + s[1]);
  if (s[2] > (s[1] + s[3]) / 2)
  {
    error = (s[3] - s[1]);
  }
  if (abs(error - lastError) > 1500)
  {
    error = lastError;
  }

  int correction = (error * Kp) + ((error - lastError) * Kd);
  lastError = error;

  int leftSpeed = baseSpeed + correction;
  int rightSpeed = baseSpeed - correction;

  if (leftSpeed > 200) leftSpeed = 200;
  if (leftSpeed < -150) leftSpeed = -150;
  if (rightSpeed > 200) rightSpeed = 200;
  if (rightSpeed < -150) rightSpeed = -150;

  motors.setSpeeds(leftSpeed, rightSpeed);
}
